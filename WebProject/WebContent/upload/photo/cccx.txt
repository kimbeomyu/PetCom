CREATE TABLE BASKET
(
    PRODUCT_NAME     VARCHAR2(500)    NOT NULL, 
    PRODUCT_PRICE    VARCHAR2(50)     NOT NULL, 
    TOTAL_PRICE      VARCHAR2(200)    NOT NULL, 
    TOTAL_DC         VARCHAR2(200)    NOT NULL, 
    ENROLL_NO        VARCHAR2(20)     NOT NULL, 
    CONSTRAINT BASKET_PK PRIMARY KEY (ENROLL_NO)
)
/

/

COMMENT ON COLUMN BASKET.PRODUCT_PRICE IS '가격'

/

COMMENT ON COLUMN BASKET.ENROLL_NO IS '상품등록번호'
